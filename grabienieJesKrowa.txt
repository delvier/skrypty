gosub OpenPaperdoll

set %ile_lootu 0
set %plecak #BACKPACKID
set %forgettime #SCNT
set #NEXTCPOSX 15
set #NEXTCPOSY 396

loop1:
  onhotkey F10 | ALT  ; loot targeted body
  {
    set #TARGCURS 1
    targetloop:
      if #TARGCURS = 1
        goto targetloop
    set #LOBJECTID #LTARGETID
    set %body #LTARGETID
    finditem %body G_2
    if #FINDKIND <> -1
    {
      if #FINDTYPE = YFM
      {
        event macro 17 0           ; open container
        gosub GumpWait container_gump NULL
        gosub LootStuff %body
        set #NEXTCPOSX 15
        set #NEXTCPOSY 396
      }
    }
    else
    {
      sound ; *bonk* when not in range!
    }
    event sysmessage Zgrabiles co Tylko sie dalo!
      sound C:WINDOWSMediading.wav ; *ding* Kiedy skonczy!
  }
  onhotkey F2  ; lotuje w okolicy 2 pol
  {
    loop2:
      finditem YFM G_2
      if #FINDKIND <> -1
      {
        set %body #FINDID
        set #LOBJECTID %body
        event macro 17 0
        gosub GumpWait container_gump NULL
        gosub LootStuff %body
        ignoreitem %body
        set #NEXTCPOSX 15
        set #NEXTCPOSY 396
        goto loop2
      }
      event sysmessage Zgrabiles co Tylko sie dalo!
      sound C:WINDOWSMediading.wav ; *ding* Kiedy skonczy!
  }
  if ( #SCNT > ( %forgettime + 120 ) )
  {
    set %forgettime #SCNT
    ignoreitem reset
  }
goto loop1

; %1 = Body ID
; %2 = Item type
sub GrabItem
  set %tempbody %1
  set %item %2

  search_again1:
    finditem %item C_ , %tempbody  ; szukaj itemow.
    if #findkind <> -1
    {
      exevent drag #FINDID #FINDSTACK
      wait 1
      exevent dropc %plecak
      repeat
       finditem %item C
      until #FINDBAGID = %plecak || #FINDKIND = -1
      goto search_again1
    }
return
;-------------------------------------------------------------------------------
; %1 = Gumpname 1
; %2 = Gumpname 2
sub GumpWait
  wait 10
  set %timedelay #SCNT
  loopwait1:
    if #CONTNAME = %1 || #CONTNAME = %2
      return
    if #SCNT > %timedelay + 7
      return
    goto loopwait1
return
;-------------------------------------------------------------------------------
sub OpenPaperdoll
  gosub GumpWait NULL paperdoll_gump NULL
  contpos 750 0
return
