event sysmessage F1 [pojedynczo] | ALT F1 [automat]
set %aktywny #false

loop:

onhotkey F1 ALT
  gosub toggleSub
onhotkey F1
  gosub goShadow
gosub stealth

goto loop
  
sub stealth

if %aktywny = #true
{
 if %lastUsed = N/A || #TIME - %lastUsed > 10
 {
  gosub goShadow
  set %lastUsed #TIME
 }
}

return

sub toggleSub

 if %aktywny = #false
 {
  event sysmessage Stealth aktywowany.
  set %aktywny #true
 }
 else
 {
  event sysmessage Stealth deaktywowany.
  set %aktywny #false
 }

return


sub goShadow
if H in #charstatus
  {
   event macro 13 47
  }
  else
  {
   event macro 13 21
  }
return
