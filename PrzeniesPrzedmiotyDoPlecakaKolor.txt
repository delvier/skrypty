; Drag przedmioty do plecaka
loop:
onhotkey F3
 gosub rzeczy
goto loop

sub rzeczy
 ignoreItem reset
 finditem ZJF C_ , #BACKPACKID
 if #findid <> x
 {
  set %sklad #FINDID
  event sysmessage Plecak: #FINDID
 }
 else
 {
  event sysmessage Nie znaleziono plecaka.
  return
 }

 repeat
  finditem FSF C_ , #BACKPACKID
  if #FINDCNT > 0
  {
   if #FINDCOL = 38 || #FINDCOL = 104
   {
    call libs\przenoszenie.lib.euo przenies #findid %sklad #findstack
    finditem FSF C_ , #BACKPACKID
   }
  }
 until #FINDCNT = 0

return
