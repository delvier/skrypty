 }

; Wyciachanie
 if 1 = 0

      gosub Przetop
      set %czas 2 * %licznik + 3

      set %licznik 0
      goto poczatek

      continue
    }

 {
  set %licznik 0
  gosub PoliczItemy

    {
      wait 10
      click 40 500 dmc

      set %licznik 0
      set %przetopione 1
      event sysmessage Pelen pojemnik. Wskaz nowy.

    }
    else

      continue

   {

          exevent dropc %KOSZ

; Koniec wyciachania

sub OtworzPojemnik
  event sysmessage Otwieram pojemnik %1
  set #lobjectid %1
  event macro 17 0
  wait 1s
return

sub ZnajdzNarzedzia

 FindItem %narzedzia_alchemika C_ , %pojemnik_z_narzedziami

   {
   display Nie moge znalesc narzedzi do majsterkowania
   halt
   }

 set %mlotek #FindID

 exevent drag %mlotek
 exevent dropc #backpackid

 event sysmessage Znalazlem mlotek: %mlotek
return

 wait 10

return

sub ScanJournal
set %__JIndex %1
set %_timeout #SCNT
ScanLoop:
if #JIndex > %__JIndex
   {
   set %__JIndex %__JIndex + 1
   ScanJournal %__JIndex
   for %__i 2 %0
       {
       if % . %__i in #journal
          return %__i
       }
   }
goto ScanLoop
return

sub SprawdzGlod
glodloop:
repeat
      event macro 8 7
      wait 5
until #ContID = #BackpackID

set %_JIndex #JIndex

gosub ScanJournal %_JIndex Twoja_postac_jest_tak_najedzona Twoja_postac_jest_najedzona_do_syta_i_nic_wiecej Twoja_postac_jest_najedzona_do_syta Twoja_postac_jest_najedzona Twoja_postac_nie_jest_glodna Twoja_postac_czuje_oznaki_glodu Twoja_postac_jest_nieco_glodna Twoja_postac_jest_glodna Twoja_postac_jest_bardzo_glodna Twoja_postac_przymiera_glodem Twoja_postac_jest_bliska_smierci_glodowej
if #Result <= 5
   return
if #Result > 5
   {
   FindItem %zarcie C
   if #FindKind = -1
      {
      event sysmessage Nie znalazlem jedzenia
      return
      }
   set #LObjectID #FindID
   event macro 17 0
   }
goto glodloop
return

sub WskazPojemnik
   event sysmessage Wskaz pojemnik albo kosz.
   set #TargCurs 1
   repeat
   until #TargCurs = 0
   set %KOSZ #LTargetID
   event sysmessage Wskazano.
return

sub IgnorujPrzedmioty
   repeat
   FindItem * C_ , #BackpackID
   if #FindKind <> -1
        {
          IgnoreItem #FindID 1
        }
   until #FindKind = -1
return

sub ZnajdzSzczypce
FindItem TBG C_ , #BackpackID
if #FindKind = -1
   {
   display Szczypce brak :<.
   halt
   }
set %szczypce #FindID
event sysmessage Znaleziono szczypce.
return

sub Przetop
set #lObjectID %szczypce
wait 10
event macro 17 0

set #LTargetID %KOSZ
wait 10
event macro 22 0
return

sub PoliczItemy

     set %licznik %licznik + 1
