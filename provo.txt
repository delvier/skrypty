; = [ Instructions ] ==============================
;--------------------------------------------------
; Personally, I found no good scripts released for
; this barding skill.  They all required you to
; select two animals and then they would repeat the
; process over and over, even if the creatures had
; died.  Well, if you've ever tried provocation then
; you know that it can be hard to stop and start a
; script while being chased by creatures.  Also, no
; script said what to use to train or when to train
; what, neither did any site I've found.  With this
; script, you go out hunting just like normal and
; when creatures get within range, you will auto -
; matically try to provoke them if they are within
; your gaining range.  This script will target
; predetermined targets that are in a 30%-75% success
; range.  You can also set the script to target
; only user defined creatures.  A table of creatures
; along with their corresponding barding difficulty
; is located at the end of the options.

; = %instruments ==================================
; This is a listing of all available instrument
; types for your shard.
; [ Recommended Value is _PGP_MQF_QRF_PRF_LQF_OQF_NRF_ ]
; PGP = Bamboo Flute
; MQF = Drums
; QRF = Lap Harp
; PRF = Lute
; LQF = Tambourine
; OQF = Tambourine with Tassel
; NRF = Standing Harp
; PRF = Iolo's Lute
; QRF = Gwenno's Harp
set %instruments _PGP_MQF_QRF_PRF_LQF_OQF_NRF_
; = Various Messages ==============================
; Below are the various messages that could be
; show during the provocation process.  For a more
; detailed explaination of each message, just look
; to the far right of each message declaration.
; [ Recommended Values are below ]
; %instrumentMessage = What_instrument_shall_you_play?
; %targetMessage = Whom_do_you_wish_to_incite?
; %retargetMessage = You_play_your_music_and_your_target_becomes_angered.__Whom_do_you_wish_them_to_attack?
; %successMessage = Your_music_succeeds,_as_you_start_a_fight.
; %failureMessage = Your_music_fails_to_incite_enough_anger.
; %musicFailureMessage = You_play_poorly,_and_there_is_no_effect.
; %distanceMessage = The_creatures_you_are_trying_to_provoke_are_too_far_away_from_each_other_for_your_music_to_have_an_effect.
; %tooFarMessage = That_is_too_far_away.
; %waitMessage = You_must_wait_a_few_moments_to_use_another_skill.
set %instrumentMessage What_instrument_shall_you_play? ; This is the message shown when you must target an instrument to use.  This occurs when no instrument is set as the "current" to use.
set %targetMessage Whom_do_you_wish_to_incite? ; This is the initial message shown when you first attempt to provoke.
set %retargetMessage You_play_your_music_and_your_target_becomes_angered.__Whom_do_you_wish_them_to_attack? ; This is the message shown after successfully targeting the first creature.
set %successMessage Your_music_succeeds,_as_you_start_a_fight. ; This is the message shown if you are successful at provoking the two creatures.
set %failureMessage Your_music_fails_to_incite_enough_anger. ; This is the message shown if you are unsuccessful at provoking the two creatures.
set %musicFailureMessage You_play_poorly,_and_there_is_no_effect. ; This is the message shown if you fail to pass the required internal musicianship check on your instrument.
set %distanceMessage The_creatures_you_are_trying_to_provoke_are_too_far_away_from_each_other_for_your_music_to_have_an_effect. ; This message is shown when the two creatures you are trying to target too far apart.  Lower your %maxDistanceToTarget value to prevent this message.
set %tooFarMessage That_is_too_far_away. ; This is the message shown when you try to target a creature too far away from you.  Lower your %maxDistanceToTarget value to prevent this message.
set %waitMessage You_must_wait_a_few_moments_to_use_another_skill. ; This message is shown when your delay value is set too low.  The script will try to prevent this message itself, however, increasing your %delay value will prevent this message without the script interferring.
set %noChanceMessage You_have_no_chance_of_provoking_those_creatures. ; This is the message shown when trying to provoke creatures that cannot be provoked.
; = %delay ========================================
; This is the time to wait between skill uses.  If
; this is set too low, the program will auto -
; calculate the correct figure to use.  It may take
; a couple skill attempts for this calculation to
; be set properly.
; [ Recommended Value is 10 ]
set %delay 10
; = %maxDistanceToTarget ==========================
; This is the furthest distance away from your
; character that you can target a creature.  It is
; also used as the furthest distance allowed
; between your targeted creatures.
; [ Recommended Value is 8 ]
set %maxDistanceToTarget 5
; = %showCountDownTimer ===========================
; This can either be #true or #false.
; When #true an estimated timer will be placed above
; your character's head showing how many seconds you
; have to wait before attempting your skill again.
; [ Recommended Value is #true ]
set %showCountDownTimer #true
; = %useUserDefinedTargets ========================
; This can either be #true or #false.
; When #true an the program WILL NOT define targets
; based on your skill value but instead will only
; use targets matching the ID's or Types declared
; in %useTheseTargets
; [ Recommended Value is #false ]
set %useUserDefinedTargets #false
; = %useTheseTargets ==============================
; You define what creatures to use as targets here.
; Seperate each creature Type or ID using an under-
; score ( _ ).  You can have as many ID's or Types
; as you like.  Keep in mind that
; %useUserDefinedTargets must be set as #true for
; these values to be used.

; = MAIN ==========================================
; This is just the main method that runs the entire
; program.  It merely checks for available targets
; and provokes them if possible.  Then its a simple
; rinse and repeat of itself.
ignoreitem #charid
chooseskill provocation
set %origSkill #skill
set %attempts 0
set %successes 0
set %occasionalTimer 0
macro:
  gosub setTargetable
  gosub setTargets
  if #result
  {
    set %attempts %attempts + 1
    gosub provo
    if #result
      set %successes %successes + 1
    chooseskill provocation
    set %gains #skill - %origSkill
    str len %gains
    set %len #strres - 1
    str left %gains %len
    set %whole #strres
    set %len %len - ( %len - 1 )
    str right %gains %len
    set %decimal #strres
    set %gains %whole , #dot , %decimal
    set %failures %attempts - %successes
  }
  if #scnt > %occasionalTimer
  {
    if %showOccasionalStats
      event sysmessage Attempts: %attempts Successes: %successes Failures: %failures Gains: %gains
    set %occasionalTimer 20 + #scnt
  }
goto macro

;
sub setTargets
{
  finditem %targetable g_ , %maxDistanceToTarget
  if #findkind
  {
    ignoreitem reset
    ignoreitem #charid
    set %target_one none
    return #false
  }
  set %target_one #findid
  ignoreitem %target_one targetid
  finditem %targetable g_ , %maxDistanceToTarget
  if #findkind
  {
    ignoreitem reset targetid
    set %target_two none
    return #false
  }
  set %target_two #findid
  gosub distance %target_one %target_two
  if #result > %maxDistanceToTarget
    return #false
  return #true
}

; = distance ======================================
; This takes two ID's or Types as parameters and
; compares their distances apart from each other.
; [ Parameters ]
; %1 = targetid of the first target
; %2 = targetid of the second target
; [ Returned Value ]
; #result will be the distance between the creatures
;
sub distance
{
  ignoreitem reset targetid
  set %one %1
  set %two %2
  finditem %one G
  ignoreitem %one creature
  set %onex #findx
  set %oney #findy
  finditem %two G
  ignoreitem %two creature
  set %twox #findx
  set %twoy #findy
  set %xdist %onex - %twox ABS
  set %ydist %oney - %twoy ABS
  if %xdist > %ydist
    set %dist %xdist
  else
    set %dist %ydist
  return %dist
}

; = provo =========================================
; This method is the heart of the script.  It uses
; the two local script variables %target_one and
; %target_two as targets while provoking.  It attemps
; the skill process and uses the character's journal
; as its method of execution procedure.
; [ Returned Value ]
; #result = #true if you successfully provoke the
;           creatures
; #result = #false if you are unsuccessful at
;           provoking the creatures due to music
;           or provocation skill failure or if
;           you run out of instruments
; #result = Timeout if the script is unable to find the
;           various journal messages needed for the
;           scripting processes.
; #result = Retarget if you either the two targets
;           are too far apart or if you are too far
;           away from your target
; #result = null if you are unlucky enough to encounter
;           an error with the sub
; [ WARNING ]
; This script WILL PAUSE if you run out of instruments.
; Place some instruments in your backpack and restart
; or unpause the script to continue.  You will ( or
; should ) also recieve this message and action if you
; do not have the container holding your instruments
; open.
sub provo
{
  set #ltargetkind 1
  startProvo:
  set %provoIndex #jindex
  event Macro 13 22
  provoCheckOne:
  set %scanTimeout 20 + #scnt
  repeat
  {
    if #jindex > %provoIndex
      set %provoIndex %provoIndex + 1
    scanjournal %provoIndex
    wait 5
  }
  until %instrumentMessage in #journal || %targetMessage in #journal || %waitMessage in #journal || %scanTimeout < #scnt
  if %scanTimeout < #scnt
    return Timeout
  if %waitMessage in #journal
  {
    set %delay %delay + 1
    wait 20
    goto startProvo
  }
  if %instrumentMessage in #journal
  {
    finditem %instruments C
    if #findkind
    {
      event sysmessage Unable to find any instruments!  Restock and unpause the script.
      pause
      return #false
    }
    set #ltargetid #findid
    event macro 22 0
    wait 30
    goto provoCheckOne
  }
  if %targetMessage in #journal
  {
    set #ltargetid %target_one
    event Macro 22 0
    set %scanTimeout 20 + #scnt
    repeat
    {
      if #jindex > %provoIndex
        set %provoIndex %provoIndex + 1
      scanjournal %provoIndex
      wait 5
    }
    until %retargetMessage in #journal || %tooFarMessage in #journal || %scanTimeout < #scnt
    if %scanTimeout < #scnt
      return Timeout
    if %tooFarMessage in #journal
    {
      set %maxDistanceToTarget %maxDistanceToTarget - 1
      return Retarget
    }
    if %retargetMessage in #journal
    {
      set #ltargetid %target_two
      event Macro 22 0
      set %scanTimeout 20 + #scnt
      repeat
      {
        if #jindex > %provoIndex
          set %provoIndex %provoIndex + 1
        scanjournal %provoIndex
        wait 5
      }
      until %successMessage in #journal || %failureMessage in #journal || %musicFailureMessage in #journal || %distanceMessage in #journal || %noChanceMessage in #journal || %scanTimeout < #scnt
      if %scanTimeout < #scnt
        return Timeout
      if %distanceMessage in #journal || %noChanceMessage in #journal
        return Retarget
      if %musicFailureMessage in #journal || %failureMessage in #journal || %successMessage in #journal
      {
        set %return #false
        if %successMessage in #journal
          set %return #true
        set %timeToReProvo #scnt2 + 10 * %delay
        while #scnt2 < %timeToReProvo
        {
          wait 20
          if %showCountDownTimer
          {
            set %showTime ( %timeToReProvo - #scnt2 ) / 10
            event exmsg #charid 3 0 %showTime
          }
        }
        return %return
      }
    }
  }
  return null ; This is just a failsafe to make sure you exit the sub incase an error occurs throughout the sub.
}


sub peace
{
  set #ltargetkind 1
  event macro 13 9
  target 2s
  set #ltargetid %target_one
  event macro 22 0
  set %wait 20 * %delay
  wait %wait
  return
}
