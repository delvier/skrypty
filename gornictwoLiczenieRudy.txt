sub skin
    if *miningInfoFirstrun = 0 || *miningInfoFirstrun = N/A
    {
     set *mainfc $FFB09F
     set *bgc $711400
     set *btnfc $711400
     set *rys $FF0099
     set *bbg $FFB09F
     set *miningInfoFirstrun 1
     goto konskin
    }
    str right #menubutton 1
    namespace local ns , #strres
    set *mainfc !bgcolor
    set *bbg !bgcolor
    set *bgc !color
    set *btnfc !color
    set *rys !rys
    konskin:
return

sub count
{
    set %pack %1
    if %open = #true && %pack <> #backpackid
    {
       set #lobjectid %pack
       event macro 17
       repeat
       until #contid = %pack
    }
    repeat
    {
       finditem %containers C_ , %pack
       if #findkind <> -1
       {
          ignoreitem #findid
          gosub count #findid %pack
          set %pack #result
       }
       finditem %containers C_ , %pack
    }
    until #findkind = -1

    repeat
    {
          finditem %ore C_ , %pack
          if #findkind <> -1
          {
             namespace local _ , #findcol
             if !ore = N/A 1
                set !ore 0
             set !ore !ore + #findstack
             ignoreitem #findid
          }
    }
    until #findkind = -1
    return %2
}

sub pointAnotherCont
