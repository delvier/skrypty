sub zmiana_miejsca
 event sysmessage Zmiana miejsca: %miejsce z *gornictwo_miejsca
 if %miejsce = 0 || %miejsce = n/a || %miejsce > *gornictwo_miejsca
  set %miejsce 1

 for %i 1 *gornictwo_miejsca
 {
  event sysmessage Miejsce %i miejsce: %miejsce
  if %miejsce = %i
  {
   repeat
    event pathfind *xpos . %i *ypos . %i
    wait 1s
   until #charposx = *xpos . %i && #charposy = *ypos . %i
   event sysmessage Jestem w miejscu: %miejsce z *gornictwo_miejsca
   break
  }
 }

return

sub odkladanie
 event sysmessage Odkladanie rzeczy

 {
  repeat
   event pathfind %pozycja_startowa_x %pozycja_startowa_y
   wait 1s
  until #charposx = %pozycja_startowa_x && #charposy = %pozycja_startowa_y
 }

 repeat
  finditem %przedmioty_do_skladu C
  if #findid <> x
  {
   exevent drag #findid #findstack
   wait %dragPauza
   exevent dropc %sklad
   wait %dragPauza
  }
 wait 2
 until #findid = x

 event macro 17 0
 wait 1s

  {
   set #lobjectid %pojemnik
   set #ltargetkind 1
   event macro 17 0
   wait 1s
   finditem ZLF C_ , %pojemnik
   exevent drag #findid 5
   wait %dragPauza
   exevent dropc #backpackid
   wait %dragPauza
 }
  
  ; Ziola
 finditem WZF C_ , #BACKPACKID
  if #findstack < %ileZiol
  {
   set #lobjectid %pojemnik
   set #ltargetkind %ileZiol
   event macro 17 0
   wait 1s
   
   finditem WZF C_ , %pojemnik
   exevent drag #findid %ileZiol
   wait %dragPauza
   exevent dropc #backpackid
   wait %dragPauza

   finditem KUF C_ , %pojemnik
   exevent drag #findid %ileZiol
   wait %dragPauza
   exevent dropc #backpackid
   wait %dragPauza
 }

sub ustaw_narzedzie
 event sysmessage Ustawiam narzedzie

 if *gornictwo_tryb = 1
 {
  finditem NPF C_ , #charid
  if #findkind = -1
  {
   finditem NPF C
   if #findkind = -1
   {
    if #charposx <> %pozycja_startowa_x || #charposy <> %pozycja_startowa_y
    {
    repeat
     event pathfind %pozycja_startowa_x %pozycja_startowa_y
     wait 1s
    until #charposx = %pozycja_startowa_x && #charposy = %pozycja_startowa_y
    }
    
    if #charposx = %pozycja_startowa_x && #charposy = %pozycja_startowa_y
    {
     if #hits < 5 || #stamina = 0
     {
      dupa:
       sound C:\WINDOWS\Media\Kopia.wav
      goto dupa
     }
     set #lobjectid %pojemnik
     event macro 17 0
     wait 1s
     
     set #lobjectid %kilofy
     event macro 17 0
     wait 1s

     finditem NPF C
     exevent drag #FINDID 1
     wait %dragPauza
     exevent dropc #backpackid 90 20
     wait %dragPauza

     gosub zmiana_miejsca

    }

   }
  set %narzedzie #findid
  exevent drag %narzedzie
  wait %dragPauza
  exevent droppd
  wait %dragPauza
 }
 else
  set %narzedzie #findid
}

sub ScanJournal
; event sysmessage ScanJournal
 set %JIndex %1
 set %_timeout #SCNT

 ScanLoop:
  set %x %timeout + %_timeout
  if #SCNT > %_timeout + %timeout
   return TIMEOUT
  if #JIndex > %JIndex
  {
   set %JIndex %JIndex + 1
   ScanJournal %JIndex
   for %i 2 %0
   {
    if % . %i in #journal
     return %i
   }
  }
 goto ScanLoop
return

sub ArmsLore
 FindItem %armslore_badaj C_ , #BackPackID
 if #FindKind <> -1
 {
  set %ALtimeDiff #TIME - %ostatnieUzycieAL
  if %ALtimeDiff > 10
  {
   event sysmessage AL time diff: %ALtimeDiff
   event macro 13 4
   set %lastTarget #FindID
   event sysmessage LT: %lastTarget
   target 3s
   set #LTargetID %lastTarget
   set #LTargetKind 1
   event macro 22 0
   ignoreitem %lastTarget
   set %ostatnieUzycieAL #TIME
  }
  else
  {
   event sysmessage AL Time diff: %ALtimeDiff
  }
 }
if #FindKind = -1
 {
  event sysmessage Reset ignorowanych przedmiotow dla ArmsLore
  ignoreItem reset
 }
}
return
